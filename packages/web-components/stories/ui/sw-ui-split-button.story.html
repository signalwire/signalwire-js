sw-ui-split-button

Variant 1 — Toggle + dropdown (sw-split-button-toggle + sw-dropup-select)

Active/inactive slots + items 
<sw-ui-split-button
  items='[
    "Option A",
    {"label":"Option B", "id":"b", "selected":true},
    "Option C"
  ]'>
  <sw-ui-icon name="camera-off" slot="inactive"></sw-ui-icon>
  <sw-ui-icon name="camera-on"  slot="active"></sw-ui-icon>
</sw-ui-split-button>

// Events
el.addEventListener('sw-split-button-toggle', (e) => console.log(e.detail)); // boolean
el.addEventListener('sw-dropup-select', (e) => console.log(e.detail)); // { id, label, selected }
